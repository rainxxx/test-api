public class Result implements Serializable{

    private int code; //返回码,0表示成功
    private String message; //返回提示消息
    private PageResult data; //返回的实体类

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public PageResult getData(){
        return data;
    }

    public void setData(PageResult data){
        this.data = data;
    }

}

public class PageResult implements Serializable{

    private long total; //总记录数
    private RewardListVO[] items; //结果集

    public long getTotal(){
        return total;
    }

    public void setTotal(long total){
        this.total = total;
    }

    public RewardListVO[] getItems(){
        return items;
    }

    public void setItems(RewardListVO[] items){
        this.items = items;
    }

}

public class RewardListVO implements Serializable{

    private long id; //奖励规则Id
    private int type; //规则类型 1.折扣 2.金额
    private String name; //规则名称
    private String description; //规则描述

    public long getId(){
        return id;
    }

    public void setId(long id){
        this.id = id;
    }

    public int getType(){
        return type;
    }

    public void setType(int type){
        this.type = type;
    }

    public String getName(){
        return name;
    }

    public void setName(String name){
        this.name = name;
    }

    public String getDescription(){
        return description;
    }

    public void setDescription(String description){
        this.description = description;
    }

}
