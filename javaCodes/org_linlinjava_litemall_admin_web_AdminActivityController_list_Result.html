public class Result implements Serializable{

    private int code; //返回码,0表示成功
    private String message; //返回提示消息
    private PageResult data; //返回的实体类

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public PageResult getData(){
        return data;
    }

    public void setData(PageResult data){
        this.data = data;
    }

}

public class PageResult implements Serializable{

    private long total; //总记录数
    private ActivityListVO[] items; //结果集

    public long getTotal(){
        return total;
    }

    public void setTotal(long total){
        this.total = total;
    }

    public ActivityListVO[] getItems(){
        return items;
    }

    public void setItems(ActivityListVO[] items){
        this.items = items;
    }

}

public class ActivityListVO implements Serializable{

    private long id; //活动Id
    private String name; //活动名称
    private int type; //活动类型 10分享 20团销 30秒杀
    private String image; //活动导航图
    private boolean isEnable; //是否已启用(上架)
    private long activityRewardRuleId; //奖励规则id
    private String typeDesc; //活动类型枚举描述(分享/团销/秒杀...)

    public long getId(){
        return id;
    }

    public void setId(long id){
        this.id = id;
    }

    public String getName(){
        return name;
    }

    public void setName(String name){
        this.name = name;
    }

    public int getType(){
        return type;
    }

    public void setType(int type){
        this.type = type;
    }

    public String getImage(){
        return image;
    }

    public void setImage(String image){
        this.image = image;
    }

    public boolean getIsEnable(){
        return isEnable;
    }

    public void setIsEnable(boolean isEnable){
        this.isEnable = isEnable;
    }

    public long getActivityRewardRuleId(){
        return activityRewardRuleId;
    }

    public void setActivityRewardRuleId(long activityRewardRuleId){
        this.activityRewardRuleId = activityRewardRuleId;
    }

    public String getTypeDesc(){
        return typeDesc;
    }

    public void setTypeDesc(String typeDesc){
        this.typeDesc = typeDesc;
    }

}
