#import <Foundation/Foundation>;

@interface Result : NSObject

//返回码,0表示成功
@property (nonatomic, strong) NSInteger code;
//返回提示消息
@property (nonatomic, assign) NSString message;
//返回的实体类
@property (nonatomic, strong) BannerVO data;


@end



#import <Foundation/Foundation>;

@interface BannerVO : NSObject

//金刚区Id
@property (nonatomic, strong) long id;
//金刚区名称
@property (nonatomic, assign) NSString name;
//金刚区图片
@property (nonatomic, assign) NSString image;
//活动类型
@property (nonatomic, strong) NSInteger activityType;
//活动Id
@property (nonatomic, strong) long activityId;
//是否上架
@property (nonatomic, strong) BOOL isEnable;
//活动类型描述[枚举对应的中文]
@property (nonatomic, assign) NSString activityTypeDesc;


@end
