#import <Foundation/Foundation>;

@interface Result : NSObject

//返回码,0表示成功
@property (nonatomic, strong) NSInteger code;
//返回提示消息
@property (nonatomic, assign) NSString message;
//返回的实体类
@property (nonatomic, strong) RewardRuleVO data;


@end



#import <Foundation/Foundation>;

@interface RewardRuleVO : NSObject

//奖励规则Id
@property (nonatomic, strong) long id;
//规则类型 1.折扣 2.金额
@property (nonatomic, strong) NSInteger type;
//规则名称
@property (nonatomic, assign) NSString name;
//奖励规则明细
@property (nonatomic, strong) ShareRewardRuleDetail[] rules;
//规则描述
@property (nonatomic, assign) NSString description;


@end



#import <Foundation/Foundation>;

@interface ShareRewardRuleDetail : NSObject

//邀请人数
@property (nonatomic, strong) NSInteger invitedCount;
//奖励 折扣or金额
@property (nonatomic, strong) double reward;


@end
